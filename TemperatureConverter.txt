program TemperatureConverter;
var
  choice: integer;
  temp, result: real;
begin
  writeln('1. Celsius to Fahrenheit');
  writeln('2. Fahrenheit to Celsius');
  write('Choice: ');
  readln(choice);
  
  write('Enter temperature: ');
  readln(temp);
  
  if choice = 1 then
    result := (temp * 9/5) + 32
  else
    result := (temp - 32) * 5/9;
  
  writeln('Converted: ', result:0:2);
end.
